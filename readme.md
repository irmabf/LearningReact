# Create package.json

  yarn init
  or
  npm init

# Install Live Server and Babel dependencies locally

  yarn add live-server babel-cli@6.24.1

# Run WebPack

  yarn run build
  or
  npm run build

# Run Live Server

  yarn run serve
  or
  npm run serve

# Run Babel

## Run Babel for the Indecision App

  yarn run build-babel
  or
  npm run build-babel
